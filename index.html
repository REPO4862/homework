            char a[100];
            int len, i, del;
            scanf("%s", a); len = strlen(a);
            for (; len > 0;) {
                scanf("%d", &del); 
                if (del > len) {
                    del = len;
                }
                else if (del <= 0) {
                    printf("0 이하의 수는 빼는것이 불가능합니다.\n"); continue;
                }
                for (i = del-1; i < len; i++) {
                    a[i] = a[i + 1];
                }
                len = strlen(a);
                printf("%s\n", a);
            }
            return 0;
        }

	}

	int main(){
		score a, b, h;
		scanf("%s %d %d", a.name, &a.kor, &a.eng);
		scanf("%s %d %d", b.name, &b.kor, &b.eng);
